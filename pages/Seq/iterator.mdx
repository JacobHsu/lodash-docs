# `_.prototype[Symbol.iterator]()`

Enables the wrapper to be iterable.

## Returns

`(Object)`: Returns the wrapper object.

## Example

import Embed from "runkit-embed-react";

export const Demo = () => {
  var code = `var _ = require("lodash");
var wrapped = _([1, 2]);
 
wrapped[Symbol.iterator]() === wrapped;
// => true
 
Array.from(wrapped);
// => [1, 2]
`;
  return (
    <div>
      <Embed source={code} />
    </div>
  );
};
