# `_.startCase([string=''])`

Converts string to start case.

## Arguments

`[string=''] (string)`: The string to convert.

## Returns

`(string)`: Returns the start cased string.

## Example

import Embed from "runkit-embed-react";

export const Demo = () => {
var code = `var _ = require("lodash");
_.startCase('--foo-bar--');
// => 'Foo Bar'
 
_.startCase('fooBar');
// => 'Foo Bar'
 
_.startCase('__FOO_BAR__');
// => 'FOO BAR'
`;
  return (
    <div>
      <Embed source={code} />
    </div>
  );
};
