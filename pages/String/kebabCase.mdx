# `_.kebabCase([string=''])`

Converts string to `kebab case`.

## Arguments

`[string=''] (string)`: The string to convert.

## Returns

`(string)`: Returns the kebab cased string.

## Example

import Embed from "runkit-embed-react";

export const Demo = () => {
  var code = `var _ = require("lodash");
_.kebabCase('Foo Bar');
// => 'foo-bar'
 
_.kebabCase('fooBar');
// => 'foo-bar'
 
_.kebabCase('__FOO_BAR__');
// => 'foo-bar'
`;
  return (
    <div>
      <Embed source={code} />
    </div>
  );
};

<Demo />
