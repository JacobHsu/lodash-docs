# `_.camelCase([string=''])`

Converts string to camel case.

## Arguments

`[string=''] (string)`: The string to convert.

## Returns

`(string)`: Returns the camel cased string.

## Example

import Embed from "runkit-embed-react";

export const Demo = () => {
  var code = `var _ = require("lodash");
_.camelCase('Foo Bar'); 
// => 'fooBar'
 
_.camelCase('--foo-bar--');
// => 'fooBar'
 
_.camelCase('__FOO_BAR__');
// => 'fooBar'
`;
  return (
    <div>
      <Embed source={code} />
    </div>
  );
};

<Demo />
