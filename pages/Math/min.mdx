# `_.min(array)`

Computes the minimum value of array. If array is empty or falsey, undefined is returned.

## Arguments

`array (Array)`: The array to iterate over.

## Returns

`(number)`: Returns the mean.

## Example

import Embed from "runkit-embed-react";

export const Demo = () => {
  var code = `var min = require("lodash/min");
min([4, 2, 8, 6]);
// => 2
 
min([]);
// => undefined
`;
  return (
    <div>
      <Embed source={code} />
    </div>
  );
};

            computed = current;
            result = value;
        }
    }
    return result;
}

    return minBy(array, (value) => value);
