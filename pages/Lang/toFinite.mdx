# `_.toFinite(value)`

Converts value to a finite number.

## Arguments

`value (*)`: The value to convert.

## Returns

`(number)`: Returns the converted number.  

## Example

import Embed from "runkit-embed-react";

export const Demo = () => {
  var code = `var toFinite = require("lodash/toFinite");
 
toFinite(3.2);
// => 3.2
 
toFinite(Number.MIN_VALUE);
// => 5e-324
 
toFinite(Infinity);
// => 1.7976931348623157e+308
 
toFinite('3.2');
// => 3.2
`;
  return (
    <div>
      <Embed source={code} />
    </div>
  );
};

<Demo />
