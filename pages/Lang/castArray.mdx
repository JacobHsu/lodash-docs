# `_.castArray(value)`

Casts value as an array if it's not one.


## Arguments
`value (*)`: The value to inspect.

## Returns
`(Array)`:Returns the cast array.

## Example

import Embed from 'runkit-embed-react';

export const Demo = () => {
var code =  `var castArray = require("lodash/castArray");
castArray(1);
// => [1]
 
castArray({ 'a': 1 });
// => [{ 'a': 1 }]
`   
  return (
    <div>
      <Embed source={code} />
    </div>
  );
};
