# `_.toNumber(value)`

Converts value to a number.

## Arguments

`value (*)`: The value to process.  

## Returns

`(number)`:  Returns the number.

## Example

import Embed from "runkit-embed-react";

export const Demo = () => {
  var code = `var toNumber = require("lodash/toNumber");
 
toNumber(3.2);
// => 3.2
 
toNumber(Number.MIN_VALUE);
// => 5e-324
 
toNumber(Infinity);
// => Infinity
 
toNumber('3.2');
// => 3.2
`;
  return (
    <div>
      <Embed source={code} />
    </div>
  );
};

<Demo />
