# `_.toArray(value)`

Converts value to an array.

## Arguments

`value (*)`: The value to convert.

## Returns

`(Array)`: Returns the converted array.

## Example

import Embed from "runkit-embed-react";

export const Demo = () => {
  var code = `var toArray = require("lodash/toArray");
 
toArray({ 'a': 1, 'b': 2 });
// => [1, 2]
 
toArray('abc');
// => ['a', 'b', 'c']
 
toArray(1);
// => []
 
toArray(null);
// => []
`;
  return (
    <div>
      <Embed source={code} />
    </div>
  );
};

<Demo />
