# `_.toLength(value)`

Converts value to an integer.  

**Note**: This method is loosely based on ToInteger.

## Arguments

`value (*)`: The value to convert.  

## Returns

`(number)`: Returns the converted integer.  

## Example

import Embed from "runkit-embed-react";

export const Demo = () => {
  var code = `var toLength = require("lodash/toLength");
 
toLength(3.2);
// => 3
 
toLength(Number.MIN_VALUE);
// => 0
 
toLength(Infinity);
// => 4294967295
 
toLength('3.2');
// => 3
`;
  return (
    <div>
      <Embed source={code} />
    </div>
  );
};

## Deep

export const DeepDemo = () => {
  var code = `
var toInteger = require("lodash/toInteger");  

const MAX_ARRAY_LENGTH = 4294967295; //  2^32 - 1

function toLength(value) {
    if (!value) {
        return 0;
    }
    value = toInteger(value);
    if (value < 0) {
        return 0;
    }
    if (value > MAX_ARRAY_LENGTH) {
        return MAX_ARRAY_LENGTH;
    }
    return value;
}

toLength(3.2);
// => 3
 
toLength(Number.MIN_VALUE);
// => 0
 
toLength(Infinity);
// => 4294967295
 
toLength('3.2');
// => 3
`;

return (

<div>
  <Embed source={code} />
</div>
); 
};
