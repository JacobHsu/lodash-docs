# `_.tail(array)`

Gets all but the first element of array..

## Arguments
`array (Array)`: The array to query.

## Returns
`(Array)`: Returns the slice of array.

## Example

import Embed from 'runkit-embed-react';

export const Demo = () => {
var code =  `var tail = require("lodash/tail");
tail([1, 2, 3]);
// => [2, 3]
`   
  return (
    <div>
      <Embed source={code} />
    </div>
  );
};
