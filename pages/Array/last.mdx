# `_.last(array)`

Gets the last element of array.

## Arguments
`array (Array)`: The array to query.

## Returns
`(*)`: Returns the last element of array.

## Example

import Embed from 'runkit-embed-react';

export const Demo = () => {
var code =  `var last = require("lodash/last");
last([1, 2, 3]);
// => 3
`   
  return (
    <div>
      <Embed source={code} />
    </div>
  );
};
