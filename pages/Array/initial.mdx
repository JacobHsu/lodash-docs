# `_.initial(array)`

Gets all but the last element of array.

## Arguments
`array (Array)`: The array to query.

## Returns
`(Array)`: Returns the slice of array.

## Example

import Embed from 'runkit-embed-react';

export const Demo = () => {
var code =  `var initial = require("lodash/initial");
initial([1, 2, 3]);
// => [1, 2]
`   
  return (
    <div>
      <Embed source={code} />
    </div>
  );
};
