# `_.head(array)`

Gets the first element of array.

## Arguments
`array (Array)`: The array to query.

## Returns
`(*)`: Returns the first element of array.

## Example

import Embed from 'runkit-embed-react';

export const Demo = () => {
var code =  `var head = require("lodash/head");
head([1, 2, 3]);
// => 1
`   
  return (
    <div>
      <Embed source={code} />
    </div>
  );
};
