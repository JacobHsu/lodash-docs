# `_.over([iteratees=[_.identity]])`

Creates a function that invokes iteratees with the arguments it receives and returns their results.

## Arguments

`[iteratees=[_.identity]] (...(Function|Function[]))`: The iteratees to invoke.

## Returns

`(Function)`: Returns the new function.

## Example

import Embed from "runkit-embed-react";

export const Demo = () => {
var code = `var _ = require("lodash");
var func = _.over([Math.max, Math.min]);
 
func(1, 2, 3, 4);
// => [4, 1]
`;
  return (
    <div>
      <Embed source={code} />
    </div>
  );
};

export const DeepDemo = () => {
var code =  `var _ = require("lodash");
function map(array, iteratee) {
    let index = -1;
    const length = array == null ? 0 : array.length;
    const result = new Array(length);

    while (++index < length) {
        result[index] = iteratee(array[index], index, array);
    }
    return result;
}
function over(iteratees) {
    return function (...args) {
        return map(iteratees, (iteratee) => iteratee.apply(this, args));
    };
}
var func = over([Math.max, Math.min]);
 
func(1, 2, 3, 4);
// => [4, 1]
`   
  return (
